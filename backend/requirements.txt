# OllamaLens Backend Dependencies
# Install: pip install -r requirements.txt

# ---------------------------------------------------------------------------
# Web Framework
# WHY fastapi[all]?  Includes uvicorn, pydantic, starlette — one install.
# ---------------------------------------------------------------------------
fastapi[all]==0.111.0
uvicorn[standard]==0.29.0          # ASGI server; [standard] adds websocket support

# ---------------------------------------------------------------------------
# HTTP Client
# WHY httpx and not requests?
# requests is synchronous. httpx has identical API but supports async/await,
# streaming, and HTTP/2 — all needed for Ollama communication.
# ---------------------------------------------------------------------------
httpx==0.27.0

# ---------------------------------------------------------------------------
# Database
# WHY sqlmodel + aiosqlite?
# sqlmodel = Pydantic + SQLAlchemy in one class (no schema duplication)
# aiosqlite = async wrapper around sqlite3 so DB writes don't block the event loop
# ---------------------------------------------------------------------------
sqlmodel==0.0.19
aiosqlite==0.20.0
sqlalchemy[asyncio]==2.0.30

# ---------------------------------------------------------------------------
# Settings
# ---------------------------------------------------------------------------
pydantic-settings==2.2.1          # reads from .env files with type validation

# ---------------------------------------------------------------------------
# Retry logic
# WHY tenacity over custom retry loops?
# tenacity handles exponential backoff, jitter, logging, and re-raise in
# ~4 decorator arguments vs. ~30 lines of manual code.
# ---------------------------------------------------------------------------
tenacity==8.3.0

# ---------------------------------------------------------------------------
# JSON Schema validation
# WHY jsonschema for runtime validation?
# Pydantic models are compile-time. jsonschema.validate() works with
# arbitrary user-supplied schemas at runtime — exactly what /chat/structured needs.
# ---------------------------------------------------------------------------
jsonschema==4.22.0

# ---------------------------------------------------------------------------
# Dev / Testing
# ---------------------------------------------------------------------------
pytest==8.2.0
pytest-asyncio==0.23.6
httpx                              # already listed; also used for TestClient
